				vetor[i]=entrada.nextInt();

			separação(vetor,tamanhodovetor);

	public static void separação(int [] num,int n) {
		int auxiliar=0;
		for(int i=0;i<num.length;i++) {
			for(int j=0;j<num.length;j++) {
				if(num[i]<num[j]) {
					auxiliar=num[i];

		}
