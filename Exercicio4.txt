package lista3;

import java.util.Scanner;

/*Somatório: Sua função deverá receber um vetor e retorna a soma de todos
elementos do vetor;*/
public class Exercício04 {

	public static void main(String[] args) {
		Scanner entrada=new Scanner(System.in);
		int [] vetor;
		System.out.println("Informe o tamanho do vetor: ");
		int tamanhodovetor=entrada.nextInt();
		vetor=new int[tamanhodovetor];
		for(int i=0;i<tamanhodovetor;i++) {
			System.out.println("Informe o "+ (i+1)+ " elemento do vetor: ");
			vetor[i]=entrada.nextInt();
		}
		System.out.println("A soma de todos os elementos do vetor é :"+somatório(vetor));
	}
	public static int somatório(int[] num) {
		int soma=0;
		for(int i = 0;i<num.length;i++) {
			soma+=num[i];
		}
		return soma;
	}

}
