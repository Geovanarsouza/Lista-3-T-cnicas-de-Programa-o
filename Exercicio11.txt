package lista3;

import java.util.Scanner;

/*Descompactação: Sua função deverá receber um vetor e gerar um vetor
descompactado. O número de repetições devem ser colocadas no novo vetor
conforme o número indicado.
Por exemplo: O vetor de entrada está indicado em verde e a saída está
destacada em azul.*/
public class Exercício11 {

	public static void main(String[] args) {
		Scanner entrada=new Scanner(System.in);
		int [] vetor;
		System.out.println("Informe o tamanho do vetor: ");
		int tamanhodovetor=entrada.nextInt();
		vetor=new int [tamanhodovetor];
		for(int i=0;i<vetor.length;i++) {
			System.out.println("Informe o "+(i+1)+" do vetor");
			vetor[i]=entrada.nextInt();
		}
		compactação(vetor);
	}
	public static void compactação(int [] num) {
		int[] novovetor;
		int cont=1;
		int aux=num[0];
		for(int i=0;i<num.length;i++) {
			if(aux!=num[i]) {
				cont++;
				aux=num[i];
				i--;
			}
		}
		novovetor=new int[cont];
		aux=num[0];
		for(int i=0,j=0;i<num.length;i++) {
			if(aux==num[i]) {
				novovetor[j]=aux;
			}
			else {
				aux=num[i];
				i--;
				j++;
			}
		}
		for(int i=0;i<novovetor.length;i++) {
			System.out.println(novovetor[i]);
		}
	}
}
