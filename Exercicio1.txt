package lista3;

import java.util.Scanner;

/*Pesquisa: Sua função deverá receber um vetor e retornar se um dado
elemento está dentro do vetor informado. Caso o elemento esteja no vetor
retornar a posição do elemento no vetor, caso não esteja retornar -100;*/
public class Exercício01 {

	public static void main(String[] args) {
		Scanner entrada=new Scanner(System.in);
		int [] vetor;
		System.out.println("Informe o tamanho do vetor: ");
		int tamanhodovetor=entrada.nextInt();
		vetor=new int[tamanhodovetor];
		for(int i=0;i<tamanhodovetor;i++) {
			System.out.println("Informe o "+(i+1)+" número do vetor: ");
			vetor[i]=entrada.nextInt();
		}
		System.out.println("Informe o número que deseja pesquisar: ");
		int numero_pesquisado=entrada.nextInt();
		pesquisa(vetor,numero_pesquisado);
	}
	public static boolean pesquisa(int []num,int n) {
		for(int i = 0;i<num.length;i++) {
			if(num[i]==n) {
				System.out.println("A posição do elemento é "+i);
				return true;
			}
			else {
				System.out.println("-100");
				return false;
			}	
		}
		return false;
		}	
	
	}
